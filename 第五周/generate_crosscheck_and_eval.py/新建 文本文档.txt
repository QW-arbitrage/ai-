# ========== Cross-Check ==========
def cross_check_flow(gold_events):
    """校验流量表中事件之间的连续性"""
    results = []
    sorted_events = sorted(gold_events, key=lambda x: x["event_id"])
    
    for i, event in enumerate(sorted_events):
        if i == 0:
            prev_capital = 0
        else:
            prev_capital = sorted_events[i-1].get("after_capital", 0)
        
        before = event.get("before_capital", 0)
        after = event.get("after_capital", 0)
        change = event.get("change_amount", 0)
        
        # 校验：after = before + change
        expected_after = before + change
        diff = after - expected_after
        
        results.append({
            "event_id": event["event_id"],
            "event_type": event["event_type"],
            "before": before,
            "after": after,
            "change": change,
            "expected_after": expected_after,
            "diff": diff,
            "status": "pass" if diff == 0 else "fail",
            "note": f"增资后资本={after}, 预期={expected_after}" if diff == 0 else f"差额={diff}"
        })
    
    return results

def cross_check_snapshot(gold_events, snapshot):
    """校验最终存量与累计流量是否匹配"""
    total_change = sum([e.get("change_amount", 0) for e in gold_events])
    total_shares = snapshot.get("total_shares", 0)
    
    ratio_sum = sum([s.get("share_ratio", 0) for s in snapshot.get("shareholders", [])])
    
    return {
        "total_capital_changes": total_change,
        "final_total_shares": total_shares,
        "ratio_sum": ratio_sum,
        "status": "pass" if abs(ratio_sum - 100) < 0.5 else "fail"
    }

# ========== 评估 ==========
def compare_gold_auto(gold, auto):
    """对比Gold和Auto结果"""
    comparisons = []
    
    gold_events = gold.get("events", [])
    auto_events = auto.get("events", [])
    
    for g in gold_events:
        g_id = g.get("event_id")
        matched = False
        for a in auto_events:
            if a.get("event_id") == g_id:
                matched = True
                # 比较关键字段
                issues = []
                if a.get("change_amount") != g.get("change_amount"):
                    issues.append(f"金额: gold={g.get('change_amount')}, auto={a.get('change_amount')}")
                if len(a.get("investors", [])) != len(g.get("investors", [])):
                    issues.append(f"投资方数量: gold={len(g.get('investors', []))}, auto={len(a.get('investors', []))}")
                comparisons.append({
                    "event_id": g_id,
                    "status": "exact_match" if not issues else "partial_match",
                    "issues": issues
                })
                break
        if not matched:
            comparisons.append({
                "event_id": g_id,
                "status": "missed",
                "issues": ["Auto未匹配到该事件"]
            })
    
    # 检查Auto是否有Gold中不存在的事件
    auto_ids = [a.get("event_id") for a in auto_events]
    gold_ids = [g.get("event_id") for g in gold_events]
    for a_id in auto_ids:
        if a_id not in gold_ids:
            comparisons.append({
                "event_id": a_id,
                "status": "extra",
                "issues": ["Auto提取了Gold中不存在的事件"]
            })
    
    return comparisons
